rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    function signedIn() {
      return request.auth != null && request.auth.token.email is string;
    }

    function email() {
      return request.auth.token.email;
    }

    function isStudent() {
      return signedIn() && email().matches('^[A-Za-z]+(\\.[A-Za-z]+)+@student\\.cga\\.school$');
    }

    function isTeacher() {
      return signedIn() && email().matches('^[A-Za-z]\\.[A-Za-z]+@cga\\.school$');
    }

    function isSuperAdmin() {
      return signedIn() && email() == 'j.smith@cga.school';
    }

    match /onlineSafetyCaseStudyResponses/{docId} {
      allow create: if isStudent() || isTeacher() || isSuperAdmin();
      allow read: if isTeacher() || isSuperAdmin();
      allow update, delete: if false;
    }
  }
}

// Enable Firebase Authentication > Sign-in method > Email/Password.
